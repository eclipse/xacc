424  template<typename LeftIndices, typename RightIndices>
425  std::tuple<
426  Tensor<array_size<LeftIndices>::value + 1,
427  DerivedTensorBackendBuilder, Scalar>,
428  Tensor<2, DerivedTensorBackendBuilder, Scalar>,
429  Tensor<array_size<RightIndices>::value + 1,
430  DerivedTensorBackendBuilder, Scalar>> svd(
431  LeftIndices& leftCutIndices, RightIndices& rightCutIndices,
432  double cutoff = 0.0) {
433 
434  static_assert (Rank == array_size<LeftIndices>::value + array_size<RightIndices>::value,
435  "Error - You must allocate all tensor indices to either the left or right side of this SVD cut.");
436 
437  // Keep track of the current shape of this Tensor
438  auto currentDims = shape->dimensions();
439 
440  // Reshape the Tensor to be Rank 2 such that the LeftIndices are combined
441  // into one index and the RightIndices are combined into another index
442  std::array<int, 2> dims;
443  int nDims = 0;
444  for (auto l : leftCutIndices) {
445  nDims += currentDims[l];
446  }
447  dims[0] = nDims;
448  nDims = 0;
449  for (auto r : rightCutIndices) {
450  nDims += currentDims[r];
451  }
452  dims[1] = nDims;
453 
454  // Reshape this tensor to be a Rank 2 tensor, this creates a new Tensor
455  Tensor<2, DerivedTensorBackendBuilder, Scalar> newRankTwoTensor =
456  reshape(dims);
457 
458  // Get a reference to the reshaped tensor
459  auto ref = newRankTwoTensor.createReference();
460 
461  // Compute the SVD of the Rank 2 tensor and
462  // return the U, V, pair.
463  auto USV = provider->svd(ref, cutoff);
464 
465  auto UMat = std::get<0>(USV);
466  auto SMat = std::get<1>(USV);
467  auto VMat = std::get<2>(USV);
468 
469  int nSingularValues = SMat.second.dimension(0);
470 
471  // Get U, S, and V new tensor shape
472  std::vector<int> uDims(array_size<LeftIndices>::value + 1), vDims(
473  array_size<RightIndices>::value + 1);
474 
475  for (int i = 0; i < array_size<LeftIndices>::value; i++) {
476  uDims[i] = currentDims[leftCutIndices[i]];
477  }
478 
479  uDims[array_size<LeftIndices>::value] = nSingularValues;
480 
481  for (int i = 0; i < array_size<RightIndices>::value; i++) {
482  vDims[i] = currentDims[rightCutIndices[i]];
483  }
484  vDims[array_size<RightIndices>::value] = nSingularValues;
485 
486  TensorShape uShape(uDims), vShape(vDims);
487  auto uRef = fire::make_tensor_reference(UMat.first.data(), uShape);
488  auto vRef = fire::make_tensor_reference(VMat.first.data(), vShape);
489  auto sRef = fire::make_tensor_reference(SMat.first.data(), SMat.second);
490 
491  // Return U and V as tensors
492  return std::make_tuple(
493  Tensor<array_size<LeftIndices>::value + 1,
494  DerivedTensorBackendBuilder, Scalar>(uRef),
495  Tensor<2, DerivedTensorBackendBuilder, Scalar>(sRef),
496  Tensor<array_size<RightIndices>::value + 1,
497  DerivedTensorBackendBuilder, Scalar>(vRef));
498  }
