276  void parse_response_header(
277  const std::shared_ptr<Response> &response) const {
278  std::string line;
279  getline(response->content, line);
280  std::size_t version_end = line.find(' ');
281  if (version_end != std::string::npos) {
282  if (5 < line.size())
283  response->http_version = line.substr(5, version_end - 5);
284  if ((version_end + 1) < line.size())
285  response->status_code = line.substr(version_end + 1,
286  line.size() - (version_end + 1) - 1);
287 
288  getline(response->content, line);
289  std::size_t param_end;
290  while ((param_end = line.find(':')) != std::string::npos) {
291  std::size_t value_start = param_end + 1;
292  if ((value_start) < line.size()) {
293  if (line[value_start] == ' ')
294  value_start++;
295  if (value_start < line.size())
296  response->header.insert(
297  std::make_pair(line.substr(0, param_end),
298  line.substr(value_start,
299  line.size() - value_start
300  - 1)));
301  }
302  if (boost::contains(line, "Content-Type")) {
303  std::vector<std::string> split;
304  boost::split(split, line, boost::is_any_of(":"));
305  boost::trim(split[1]);
306  response->contentType = split[1];
307  }
308  getline(response->content, line);
309  }
