231  void buildFluxMaps() {
232  // These tempInt blocks will become fPlusMap and fMinusMap eventually.
233  const size_t tempIntSize = numSpecies * numReactions / 2;
234  vector<int> tempInt1(tempIntSize,0);
235  vector<int> tempInt2(tempIntSize,0);
236 
237  // Access elements by reacMask[speciesIndex + species * reactionIndex].
238  vector<int> reacMask(numSpecies * numReactions,0); // [species][reactions]
239  vector<int> numFluxPlus(numSpecies,0), numFluxMinus(numSpecies,0);
240 
241  // Start of Guidry's original parseF() code
242  int incrementPlus = 0;
243  int incrementMinus = 0;
244  int localNumFPlus = 0, localNumFMinus = 0;
245  numFPlus = 0; numFMinus = 0;
246 
247  // Loop over all isotopes in the network
248 
249  for (int i = 0; i < numSpecies; i++) {
250  int total = 0;
251  localNumFPlus = 0;
252  localNumFMinus = 0;
253 
254  // Loop over all possible reactions for this isotope, finding those that
255  // change its population up (contributing to F+) or down (contributing
256  // to F-).
257  Species currentSpecies = species->at(i);
258  for (int j = 0; j < numReactions; j++) {
259  int totalL = 0;
260  int totalR = 0;
261 
262  // Loop over reactants for this reaction
263  Reaction & reaction = reactions->at(j);
264  for (int k = 0; k < reaction.numReactants; k++) {
265  if (currentSpecies.atomicNumber == reaction.reactantZ[k]
266  && currentSpecies.neutronNumber
267  == reaction.reactantN[k])
268  totalL++;
269  }
270 
271  // Loop over products for this reaction
272  for (int k = 0; k < reaction.numProducts; k++) {
273  if (currentSpecies.atomicNumber == reaction.productZ[k]
274  && currentSpecies.neutronNumber
275  == reaction.productN[k])
276  totalR++;
277  }
278 
279  total = totalL - totalR;
280 
281  if (total > 0) // Contributes to F- for this isotope
282  {
283  localNumFMinus++;
284  reacMask[i + numSpecies * j] = -total;
285  tempInt2[incrementMinus + localNumFMinus - 1] = j;
286  } else if (total < 0) // Contributes to F+ for this isotope
287  {
288  localNumFPlus++;
289  reacMask[i + numSpecies * j] = -total;
290  tempInt1[incrementPlus + localNumFPlus - 1] = j;
291  } else // Does not contribute to flux for this isotope
292  {
293  reacMask[i + numSpecies * j] = 0;
294  }
295  }
296 
297  // Keep track of the total number of F+ and F- terms in the network for all isotopes
298  numFPlus += localNumFPlus;
299  numFMinus += localNumFMinus;
300 
301  numFluxPlus[i] = localNumFPlus;
302  numFluxMinus[i] = localNumFMinus;
303 
304  incrementPlus += localNumFPlus;
305  incrementMinus += localNumFMinus;
306 
307  // if (showParsing == 1)
308  // printf("%d %s numF+ = %d numF- = %d\n", i, isoLabel[i], numFplus, numFminus);
309  }
310 
311  // Create 1D arrays that will be used to map finite F+ and F- to the Flux array.
312  int FplusIsotopeCut[numSpecies];
313  int FminusIsotopeCut[numSpecies];
314  int FplusIsotopeIndex[numFPlus];
315  int FminusIsotopeIndex[numFMinus];
316 
317  FplusIsotopeCut[0] = numFluxPlus[0];
318  FminusIsotopeCut[0] = numFluxMinus[0];
319  for (int i = 1; i < numSpecies; i++) {
320  FplusIsotopeCut[i] = numFluxPlus[i] + FplusIsotopeCut[i - 1];
321  FminusIsotopeCut[i] = numFluxMinus[i] + FminusIsotopeCut[i - 1];
322  }
323 
324  int currentIso = 0;
325  for (int i = 0; i < numFPlus; i++) {
326  FplusIsotopeIndex[i] = currentIso;
327  if (i == (FplusIsotopeCut[currentIso] - 1))
328  currentIso++;
329  }
330 
331  currentIso = 0;
332  for (int i = 0; i < numFMinus; i++) {
333  FminusIsotopeIndex[i] = currentIso;
334  if (i == (FminusIsotopeCut[currentIso] - 1))
335  currentIso++;
336  }
337 
338  fPlusMap.resize(numFPlus);
339  for (int i = 0; i < numFPlus; i++) {
340  fPlusMap[i] = tempInt1[i];
341  }
342  fMinusMap.resize(numFMinus);
343  for (int i = 0; i < numFMinus; i++) {
344  fMinusMap[i] = tempInt2[i];
345  }
346 
347  // Populate the fPlusMin and FplusMax arrays
348  unsigned short fPlusMin [numSpecies];
349  unsigned short fMinusMin [numSpecies];
350 
351  fPlusMin[0] = 0;
352  fPlusMaximums.resize(numSpecies);
353  fPlusMaximums[0] = numFluxPlus[0] - 1;
354  for (int i = 1; i < numSpecies; i++) {
355  fPlusMin[i] = fPlusMaximums[i - 1] + 1;
356  fPlusMaximums[i] = fPlusMin[i] + numFluxPlus[i] - 1;
357  }
358  // Populate the fMinusMin and FminusMax arrays
359  fMinusMin[0] = 0;
360  fMinusMaximums.resize(numSpecies);
361  fMinusMaximums[0] = numFluxMinus[0] - 1;
362  for (int i = 1; i < numSpecies; i++) {
363  fMinusMin[i] = fMinusMaximums[i - 1] + 1;
364  fMinusMaximums[i] = fMinusMin[i] + numFluxMinus[i] - 1;
365  }
366 
367  // Allocate the flux vectors
368  fPlusFactors.resize(numSpecies*numReactions);
369  fMinusFactors.resize(numSpecies*numReactions);
370 
371  // Populate the fPlusFac and FminusFac arrays that hold the factors counting the
372  // number of occurences of the species in the reaction. Note that this can only
373  // be done after parseF() has been run to give reacMask[i][j].
374  int tempCountPlus = 0;
375  int tempCountMinus = 0;
376  for (int i = 0; i < numSpecies; i++) {
377  for (int j = 0; j < numReactions; j++) {
378  if (reacMask[i + numSpecies * j] > 0) {
379  fPlusFactors[tempCountPlus] = reacMask[i + numSpecies * j];
380  tempCountPlus++;
381  } else if (reacMask[i + numSpecies * j] < 0) {
382  fMinusFactors[tempCountMinus] = -1.0*reacMask[i
383  + numSpecies * j];
384  tempCountMinus++;
385  }
386  }
387  } // If you run this through the debugger, it is a lot of work to calculate 0!
388 
389  // Notice that fPlusFactors == fMinusFactors! ~ Question for Mike!
390 
391  return;
392  }

413  void computeRates(const double & temp) {
414  // Compute the temperatures
415  array<double,6> tempValues;
416  double cbrtT = cbrt(temp); // Cube root of T
417  tempValues[0] = 1 / temp; // 1/T
418  tempValues[1] = 1 / cbrtT;
419  tempValues[2] = cbrtT;
420  tempValues[3] = temp;
421  tempValues[4] = cbrtT * cbrtT * cbrtT * cbrtT * cbrtT; // T^(5/3)
422  tempValues[5] = log(temp);
423 
424  // Compute the rates
425  for (Reaction & reaction : *reactions) {
426  reaction.setRate(tempValues);
427  }
428  }

435  void computeFluxes() {
436 
437  vector<double> reactionFlux(numReactions, 0);
438 
439  // Compute the flux due to each reaction
440  for (int i = 0; i < numReactions; i++) {
441  Reaction & reaction = reactions->at(i);
442  // This computes the flux as the product of the rate times some
443  // power of the current abundance. It is written as presented below
444  // to (hopefully) streamline the performance. There is a branchless
445  // implementation too, but it isn't clear to me that it is faster.
446  // 1 Body
447  reactionFlux[i] = reaction.rate
448  * (species->at(reaction.reactants[0]).massFraction);
449  // 2 Body
450  if (reaction.numReactants > 1)
451  reactionFlux[i] *=
452  (species->at(reaction.reactants[1]).massFraction);
453  // 3 Body
454  else if (reaction.numReactants > 2)
455  reactionFlux[i] *=
456  (species->at(reaction.reactants[1]).massFraction)
457  * (species->at(reaction.reactants[2]).massFraction);
458  }
459 
460  // Populate the incoming (fPlus) and outgoing flux (fMinus).
461  vector<double> fPlus(numFPlus, 0);
462  for (int i = 0; i < numFPlus; i++) {
463  fPlus[i] = fPlusFactors[i] * reactionFlux[fPlusMap[i]];
464  }
465  vector<double> fMinus(numFMinus, 0);
466  for (int i = 0; i < numFMinus; i++) {
467  fMinus[i] = fMinusFactors[i] * reactionFlux[fMinusMap[i]];
468  }
469 
470  // Sum the F+ and F- for each isotope. These are "sub-arrays"
471  // of Fplus and Fminus at (F[+ or -] + minny) of size fPlusMaximums[i].
472  int minny;
473  vector<double> fPlusSum(numSpecies, 0);
474  vector<double> fMinusSum(numSpecies, 0);
475  for (int i = 0; i < numSpecies; i++) {
476  minny = (i > 0) ? fPlusMaximums[i - 1] + 1 : 0;
477  // Sum F+.
478  fPlusSum[i] = 0.0;
479  for (int j = minny; j <= fPlusMaximums[i]; j++) {
480  fPlusSum[i] += fPlus[j];
481  }
482  // Sum F-.
483  minny = (i > 0) ? fMinusMaximums[i - 1] + 1 : 0;
484  fMinusSum[i] = 0.0;
485  for (int j = minny; j <= fMinusMaximums[i]; j++) {
486  fMinusSum[i] += fMinus[j];
487  }
488  species->at(i).flux = fPlusSum[i] - fMinusSum[i];
489  }
490 
491  return;
492  }
