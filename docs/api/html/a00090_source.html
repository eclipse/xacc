132  double getAverage() const {
133  //std::assert(measurements.size()>0);
134  std::stringstream ss;
135  double aver = 0.;
136  long n_measurements = measurements.size();
137  unsigned long tmp;
138  bool odd;
139  for (unsigned long bucket = 0; bucket < (1UL << measurements[0].size());
140  ++bucket) {
141  long count = 0; // use bucket to "collect"(i.e. count)
142  for (const auto outcome : measurements) {
143  if (outcome.to_ulong() == bucket) {
144  count++;
145  }
146  }
147  double p = double(count) / n_measurements;
148  ss << "p= " << p << std::endl;
149  tmp = bucket;
150  odd = false;
151  while (tmp) {
152  odd = !odd;
153  tmp = tmp & (tmp - 1);
154  }
155 
156  if (!odd) {
157  aver += p;
158  } else {
159  aver -= p;
160  }
161  }
162 // XACCInfo(ss.str());
163  return aver;
164  }
